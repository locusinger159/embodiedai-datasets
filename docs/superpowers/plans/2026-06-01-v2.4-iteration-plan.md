# v2.4 迭代实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为具身智能数据集导航站增加英文版、技术博客板块、数据格式对比增强和标准草案页。

**Architecture:** 保持零依赖纯静态构建。在现有 build.cjs 基础上扩展 i18n 支持（一次构建输出中英双语），新增 blog.json 数据源驱动博客系统，格式对比功能复用现有 compare modal，标准草案页为独立静态模板。

**Tech Stack:** Node.js 原生 API（fs, path），纯 HTML + CSS + 原生 JavaScript，无任何 npm 依赖。

- [ ] **Step 2: 创建 i18n 构建函数 `buildPageI18n`**

在 `buildAll` 函数内部，创建替代原 `buildPage` 的国际化版本：

```js
  function buildPage(templateFile, replacements) {
    let html = fs.readFileSync(templateFile, 'utf8');
    const headWithMeta = head.replace('{{META}}', replacements.meta || '');
    // 调整资源路径（英文站需要 ../ 前缀访问根目录资源）
    const headAdjusted = isEn
      ? headWithMeta.replace(/href="\//g, 'href="/').replace(/src="\//g, 'src="/')
      : headWithMeta;
    html = html.replace('{{HEAD}}', headAdjusted);
    html = html.replace('{{NAVBAR}}', replacements.nav || navPartial);
    html = html.replace('{{FOOTER}}', replacements.foot || footerPartial);
    for (const [key, val] of Object.entries(replacements)) {
      if (key === 'meta' || key === 'nav' || key === 'foot') continue;
      html = html.replace(new RegExp('\\{\\{' + key + '\\}\\}', 'g'), val);
    }
    return html;
  }
```

```js
  // ── Format panorama ──────────────────────────────────────
  function buildFormatData(dss) {
    const storageCounts = {};
    const sensorCounts = {};
    let standardCount = 0;
    let schemaDepths = {};

    dss.forEach(d => {
      const df = d.dataFormat || {};
      const dc = d.dataContent || {};

      // Storage format
      const storage = df.storage || 'Unknown';
      storageCounts[storage] = (storageCounts[storage] || 0) + 1;

      // Schema depth (count → separators)
      const schema = df.schema || '';
      const depth = schema.split('→').length;
      schemaDepths[depth] = (schemaDepths[depth] || 0) + 1;

      // Standard format check
      if (['RLDS', 'TFDS', 'HDF5'].some(s => (df.storage || '').includes(s))) {
        standardCount++;
      }

      // Sensors
      (dc.sensors || d.modality || []).forEach(s => {
        sensorCounts[s] = (sensorCounts[s] || 0) + 1;
      });
    });

    return {
      totalDatasets: dss.length,
      storageCounts,
      sensorCounts,
      standardFormatRatio: Math.round(standardCount / dss.length * 100),
      schemaDepths,
      // Simplified sensor matrix: top sensors by robot type
      sensorMatrix: buildSensorMatrix(dss)
    };
  }

  function buildSensorMatrix(dss) {
    const robotTypes = [...new Set(dss.flatMap(d => d.robotType || []))].sort();
    const allSensors = [...new Set(dss.flatMap(d => d.dataContent?.sensors || d.modality || []))].sort();
    // Limit to top types and sensors for readability
    const topTypes = robotTypes.slice(0, 10);
    const topSensors = allSensors.slice(0, 10);
    const rows = topTypes.map(rt => ({
      label: rt,
      sensors: dss.filter(d => (d.robotType || []).includes(rt))
        .flatMap(d => d.dataContent?.sensors || d.modality || [])
        .filter(s => topSensors.includes(s))
    }));
    return { rows, cols: topSensors };
  }
```
